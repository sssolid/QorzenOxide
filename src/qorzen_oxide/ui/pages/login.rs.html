15/// Login page component

17pub fn Login() -> Element {
18    let app_state = use_app_state();
19    let login = use_login();
20    let navigator = use_navigator();
21
22    // Form state
23    let mut username = use_signal(String::new);
24    let mut password = use_signal(String::new);
25    let mut remember_me = use_signal(|| false);
26    let mut login_error = use_signal(|| None::<String>);
27
28    // Redirect if already authenticated
29    use_effect({
30        move || {
31            if app_state.current_user.is_some() {
32                navigator.push(Route::Dashboard {});
33            }
34        }
35    });
36
37    let handle_submit = {
38        move |_| {
39            let credentials = Credentials::Password {
40                username: username(),
41                password: password(),
42            };
43
44            login_error.set(None);
45
46            // Validate form
47            if username().trim().is_empty() {
48                login_error.set(Some("Username is required".to_string()));
49                return;
50            }
51
52            if password().trim().is_empty() {
53                login_error.set(Some("Password is required".to_string()));
54                return;
55            }
56
57            // Attempt login
58            spawn({
59                let navigator = navigator;
60                async move {
61                    login.call(credentials);
62
63                    // Wait a bit for state to update, then navigate
64                    #[cfg(not(target_arch = "wasm32"))]
65                    tokio::time::sleep(std::time::Duration::from_millis(1500)).await;

70                }
