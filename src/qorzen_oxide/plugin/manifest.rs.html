123    /// Load plugin manifest from TOML file (native only)
124    #[cfg(not(target_arch = "wasm32"))]
125    pub async fn load_from_file(path: &std::path::Path) -> Result<Self> {
126        use tokio::fs;
127
128        let content = fs::read_to_string(path).await.map_err(|e| {
129            Error::file(
130                path.display().to_string(),
131                crate::error::FileOperation::Read,
132                format!("Failed to read manifest file: {}", e),
133            )
134        })?;
135
136        let manifest: PluginManifest = toml::from_str(&content).map_err(|e| {
137            Error::new(
138                crate::error::ErrorKind::Serialization,
139                format!("Failed to parse manifest TOML: {}", e),
140            )
141        })?;
142
143        manifest.validate()?;
144        Ok(manifest)
145    }

159    /// Load plugin manifest from platform file system
160    pub async fn load_from_platform(
161        path: &str,
162        filesystem: &dyn crate::platform::filesystem::FileSystemProvider,
163    ) -> Result<Self> {
164        let content_bytes = filesystem.read_file(path).await.map_err(|e| {
165            Error::file(
166                path.to_string(),
167                crate::error::FileOperation::Read,
168                format!("Failed to read manifest file via platform: {}", e),
169            )
170        })?;
171
172        let content = String::from_utf8(content_bytes).map_err(|e| {
173            Error::new(
174                crate::error::ErrorKind::Serialization,
175                format!("Invalid UTF-8 in manifest file: {}", e),
176            )
177        })?;
178
179        Self::load_from_str(&content)
180    }

195    /// Validate the manifest for correctness
196    pub fn validate(&self) -> Result<()> {
197        // Validate plugin ID format
198        if !self
199            .plugin
200            .id
201            .chars()
202            .all(|c| c.is_alphanumeric() || c == '_' || c == '-')
203        {
204            return Err(Error::plugin(
205                &self.plugin.id,
206                "Plugin ID must contain only alphanumeric characters, underscores, and hyphens",
207            ));
208        }
209
210        if self.plugin.id.is_empty() {
211            return Err(Error::plugin(&self.plugin.id, "Plugin ID cannot be empty"));
212        }
213
214        // Validate version format
215        if self.plugin.version.is_empty() {
216            return Err(Error::plugin(
217                &self.plugin.id,
218                "Plugin version cannot be empty",
219            ));
220        }
221
222        if !self.plugin.version.chars().any(|c| c.is_numeric()) {
223            return Err(Error::plugin(
224                &self.plugin.id,
225                "Plugin version must contain at least one number",
226            ));
227        }
228
229        // Validate API version compatibility
230        let core_api_version = env!("CARGO_PKG_VERSION");
231        if !self.is_api_compatible(core_api_version) {
232            return Err(Error::plugin(
233                &self.plugin.id,
234                format!(
235                    "Plugin API version {} is not compatible with core version {}",
236                    self.plugin.api_version, core_api_version
237                ),
238            ));
239        }
240
241        // Validate build configuration
242        if self.build.entry.is_empty() {
243            return Err(Error::plugin(
244                &self.plugin.id,
245                "Build entry point cannot be empty",
246            ));
247        }
248
249        // Validate search configuration if present
250        if let Some(ref search_config) = self.search {
251            for provider in &search_config.providers {
252                if provider.id.is_empty() {
253                    return Err(Error::plugin(
254                        &self.plugin.id,
255                        "Search provider ID cannot be empty",
256                    ));
257                }
258                if provider.name.is_empty() {
259                    return Err(Error::plugin(
260                        &self.plugin.id,
261                        "Search provider name cannot be empty",
262                    ));
263                }
264            }
265        }
266
267        Ok(())
268    }

270    /// Check if plugin is compatible with target platform
271    pub fn is_platform_compatible(&self, platform: &str) -> bool {
272        if self.targets.is_empty() {
273            return true; // No platform restrictions
274        }
275
276        self.targets
277            .values()
278            .any(|target| target.platform == platform || target.platform == "all")
279    }

281    /// Check if API version is compatible
282    pub fn is_api_compatible(&self, core_version: &str) -> bool {
283        // Simple semantic version check
284        let plugin_parts: Vec<&str> = self.plugin.api_version.split('.').collect();
285        let core_parts: Vec<&str> = core_version.split('.').collect();
286
287        if plugin_parts.len() < 2 || core_parts.len() < 2 {
288            return false;
289        }
290
291        // Major version must match, minor version can be equal or lower
292        plugin_parts[0] == core_parts[0]
293            && plugin_parts[1].parse::<u32>().unwrap_or(0)
294                <= core_parts[1].parse::<u32>().unwrap_or(0)
295    }

304    /// Get required permissions as Permission structs
305    pub fn get_required_permissions(&self) -> Vec<Permission> {
306        self.permissions
307            .iter()
308            .filter_map(|perm_str| {
309                let parts: Vec<&str> = perm_str.split('.').collect();
310                if parts.len() == 2 {
311                    Some(Permission {
312                        resource: parts[0].to_string(),
313                        action: parts[1].to_string(),
314                        scope: crate::auth::PermissionScope::Global,
315                    })
316                } else {
317                    None
318                }
319            })
320            .collect()
321    }
