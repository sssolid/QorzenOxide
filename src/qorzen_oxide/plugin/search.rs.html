13/// Search query structure

15pub struct SearchQuery {

19    pub limit: Option<usize>,

201    provider_health: Arc<RwLock<HashMap<String, ProviderHealth>>>,

214    /// Register a search provider from a plugin
215    pub async fn register_provider(&self, provider: Arc<dyn SearchProvider>) -> Result<()> {
216        let provider_id = provider.provider_id().to_string();
217
218        // Validate the provider
219        let health = provider.health_check().await.unwrap_or(ProviderHealth {
220            is_healthy: false,
221            response_time_ms: None,
222            error_message: Some("Health check failed".to_string()),
223            last_check: chrono::Utc::now(),
224        });
225
226        // Register the provider
227        self.providers
228            .write()
229            .await
230            .insert(provider_id.clone(), provider);
231        self.provider_health
232            .write()
233            .await
234            .insert(provider_id, health);
235
236        Ok(())
237    }

239    /// Unregister a search provider
240    pub async fn unregister_provider(&self, provider_id: &str) -> Result<()> {
241        self.providers.write().await.remove(provider_id);
242        self.provider_health.write().await.remove(provider_id);
243        Ok(())
244    }

246    /// Perform a federated search across all providers
247    pub async fn search(&self, query: SearchQuery) -> Result<SearchResponse> {
248        let start_time = std::time::Instant::now();
249        let providers = self.providers.read().await;
250
251        let mut all_results = Vec::new();
252        let mut all_facets = Vec::new();
253        let mut all_suggestions = Vec::new();
254        let mut sources = Vec::new();
255
256        // Search all providers concurrently
257        let search_tasks: Vec<_> = providers
258            .values()
259            .map(|provider| {
260                let provider = Arc::clone(provider);
261                let query = query.clone();
262                async move {
263                    let provider_id = provider.provider_id().to_string();
264                    match provider.search(&query).await {
265                        Ok(results) => Some((provider_id, results)),
266                        Err(e) => {
267                            tracing::warn!(
268                                "Search provider {} failed: {}",
269                                provider.provider_id(),
270                                e
271                            );
272                            None
273                        }
274                    }
275                }
276            })
277            .collect();
278
279        // Execute all searches
280        let search_results = futures::future::join_all(search_tasks).await;
281
282        // Collect results
283        for (provider_id, mut results) in search_results.into_iter().flatten() {
284            sources.push(provider_id);
285            all_results.append(&mut results);
286        }
287
288        // Get facets if requested
289        if !query.facets.is_empty() {
290            let facet_tasks: Vec<_> = providers
291                .values()
292                .filter(|p| p.supports_facets())
293                .map(|provider| {
294                    let provider = Arc::clone(provider);
295                    let query = query.clone();
296                    async move { provider.get_facets(&query).await.unwrap_or_default() }
297                })
298                .collect();
299
300            let facet_results = futures::future::join_all(facet_tasks).await;
301            for mut facets in facet_results {
302                all_facets.append(&mut facets);
303            }
304        }
305
306        // Get suggestions if requested
307        if query.include_suggestions {
308            let suggestion_tasks: Vec<_> = providers
309                .values()
310                .filter(|p| p.supports_suggestions())
311                .map(|provider| {
312                    let provider = Arc::clone(provider);
313                    let query = query.clone();
314                    async move { provider.get_suggestions(&query).await.unwrap_or_default() }
315                })
316                .collect();
317
318            let suggestion_results = futures::future::join_all(suggestion_tasks).await;
319            for mut suggestions in suggestion_results {
320                all_suggestions.append(&mut suggestions);
321            }
322        }
323
324        // Sort results by score (descending)
325        all_results.sort_by(|a, b| {
326            b.score
327                .partial_cmp(&a.score)
328                .unwrap_or(std::cmp::Ordering::Equal)
329        });
330
331        // Apply pagination
332        let total_count = all_results.len();
333        if let (Some(offset), Some(limit)) = (query.offset, query.limit) {
334            all_results = all_results.into_iter().skip(offset).take(limit).collect();
335        } else if let Some(limit) = query.limit {
336            all_results.truncate(limit);
337        }
338
339        // Sort suggestions by score
340        all_suggestions.sort_by(|a, b| {
341            b.score
342                .partial_cmp(&a.score)
343                .unwrap_or(std::cmp::Ordering::Equal)
344        });
345        all_suggestions.truncate(10); // Limit suggestions
346
347        let query_time = start_time.elapsed().as_millis() as u64;
348
349        Ok(SearchResponse {
350            results: all_results,
351            total_count,
352            facets: all_facets,
353            suggestions: all_suggestions,
354            query_time_ms: query_time,
355            sources,
356        })
357    }

359    /// Index content across relevant providers
360    pub async fn index_content(&self, content: IndexableContent) -> Result<()> {
361        let providers = self.providers.read().await;
362
363        let index_tasks: Vec<_> = providers
364            .values()
365            .map(|provider| {
366                let provider = Arc::clone(provider);
367                let content = content.clone();
368                async move {
369                    if let Err(e) = provider.index_content(&content).await {
370                        tracing::warn!(
371                            "Failed to index content in provider {}: {}",
372                            provider.provider_id(),
373                            e
374                        );
375                    }
376                }
377            })
378            .collect();
379
380        futures::future::join_all(index_tasks).await;
381        Ok(())
382    }

486    async fn search(&self, query: &SearchQuery) -> Result<Vec<SearchResult>> {
487        let content = self.indexed_content.read().await;
488        let query_lower = query.query.to_lowercase();
489
490        let mut results = Vec::new();
491
492        for item in content.iter() {
493            let title_match = item.title.to_lowercase().contains(&query_lower);
494            let body_match = item
495                .body
496                .as_ref()
497                .map(|b| b.to_lowercase().contains(&query_lower))
498                .unwrap_or(false);
499
500            if title_match || body_match {
501                let score = if title_match { 0.9 } else { 0.6 };
502
503                results.push(SearchResult {
504                    id: item.id.clone(),
505                    result_type: item.content_type.clone(),
506                    title: item.title.clone(),
507                    description: item.body.clone(),
508                    score,
509                    url: Some(format!("/content/{}", item.id)),
510                    thumbnail: None,
511                    metadata: item.metadata.clone(),
512                    source_plugin: self.id.clone(),
513                    timestamp: item.updated_at,
514                });
515            }
516        }
517
518        Ok(results)
519    }

521    async fn get_suggestions(&self, query: &SearchQuery) -> Result<Vec<SearchSuggestion>> {
522        let content = self.indexed_content.read().await;
523        let query_lower = query.query.to_lowercase();
524
525        let mut suggestions = Vec::new();
526
527        for item in content.iter() {
528            if item.title.to_lowercase().starts_with(&query_lower) {
529                suggestions.push(SearchSuggestion {
530                    text: query.query.clone(),
531                    completion: item.title.clone(),
532                    category: Some(item.content_type.clone()),
533                    score: 0.8,
534                });
535            }
536        }
537
538        suggestions.truncate(5);
539        Ok(suggestions)
540    }

542    async fn index_content(&self, content: &IndexableContent) -> Result<()> {
543        let mut indexed = self.indexed_content.write().await;
544
545        // Remove existing content with same ID
546        indexed.retain(|item| item.id != content.id);
547
548        // Add new content
549        indexed.push(content.clone());
550
551        Ok(())
552    }
